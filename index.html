        // --- Fixed Parameters (used in JS and some in Assumptions Table) ---

        const BUSINESS_DAYS_PER_MONTH = 21; // Still used in JS, shown in assumptions

        const EFFICIENCY_INC_LEARNING_CURVE = 1.26; // Still used in JS, shown in assumptions

        // WIN_RATE and AVG_PRODUCT_VALUE_PER_LINE are now inputs


        const PROJECTION_QUARTERS = 4; 


        // --- Chart Variable ---

        let cumulativeChartInstance = null;


        // Helper function to format numbers as currency

        function formatCurrency(value) {

            if (isNaN(value)) return '$0.00';

            return '$' + value.toFixed(2).replace(/\d(?=(\d{3})+\.)/g, '$&,');

        }


        // Helper function to format numbers as percentage

        function formatPercentage(value) {

            if (isNaN(value)) return '0.0%';

            return value.toFixed(1) + '%';

        }

        

        // Function to initialize or update the cumulative chart

        function createOrUpdateCumulativeChart(cumulativeCosts, cumulativeGrossProfit) {

            const ctx = document.getElementById('cumulativeBenefitChart').getContext('2d');

            const labels = Array.from({ length: PROJECTION_QUARTERS }, (_, i) => `Quarter ${i + 1}`);


            if (cumulativeChartInstance) {

                cumulativeChartInstance.data.labels = labels;

                cumulativeChartInstance.data.datasets[0].data = cumulativeCosts;

                cumulativeChartInstance.data.datasets[1].data = cumulativeGrossProfit;

                cumulativeChartInstance.update();

            } else {

                cumulativeChartInstance = new Chart(ctx, {

                    type: 'line',

                    data: {

                        labels: labels,

                        datasets: [

                            {

                                label: 'Cumulative Costs',

                                data: cumulativeCosts,

                                borderColor: 'rgba(239, 68, 68, 1)', 

                                backgroundColor: 'rgba(239, 68, 68, 0.1)',

                                tension: 0.1,

                                fill: false,

                                yAxisID: 'y',

                            },

                            {

                                label: 'Cumulative New Gross Profit',

                                data: cumulativeGrossProfit,

                                borderColor: 'rgba(34, 197, 94, 1)', 

                                backgroundColor: 'rgba(34, 197, 94, 0.1)',

                                tension: 0.1,

                                fill: false, 

                                yAxisID: 'y',

        // Function to calculate ROI and update chart

        function calculateROI() {

            document.getElementById('error_message').textContent = '';


            // --- Read all user inputs ---

            const numQuoters = parseFloat(document.getElementById('num_quoters').value);

            const quotesPerDayPerQuoter = parseFloat(document.getElementById('quotes_per_day_per_quoter').value);

            const linesPerQuote = parseFloat(document.getElementById('lines_per_quote').value);

            const approxGrossMarginPercent = parseFloat(document.getElementById('approx_gross_margin_percent').value);

            const winRatePercent = parseFloat(document.getElementById('win_rate_percent').value); 

            const avgProductValuePerLine = parseFloat(document.getElementById('avg_product_value_per_line').value); 


            const inputs = [numQuoters, quotesPerDayPerQuoter, linesPerQuote, approxGrossMarginPercent, winRatePercent, avgProductValuePerLine];

            if (inputs.some(isNaN) || inputs.some(val => val < 0)) {

                document.getElementById('error_message').textContent = 'Please enter valid, non-negative numbers for all fields.';

                // Reset results to 0 or default

                document.getElementById('net_new_sales_per_month').textContent = formatCurrency(0);

                document.getElementById('new_gross_profit_per_month').textContent = formatCurrency(0);

                document.getElementById('total_monthly_aq_fees').textContent = formatCurrency(FIXED_TOTAL_MONTHLY_AQ_FEES); // Display fixed fees in results

                document.getElementById('net_monthly_benefit').textContent = formatCurrency(0 - FIXED_TOTAL_MONTHLY_AQ_FEES);

                document.getElementById('full_year_roi_percent').textContent = formatPercentage(FIXED_TOTAL_MONTHLY_AQ_FEES > 0 ? -100 : 0);

                

                const zeroData = Array(PROJECTION_QUARTERS).fill(0);

                createOrUpdateCumulativeChart(zeroData, zeroData);

                return;

            }


            // Convert win rate from percentage to decimal for calculation

            const winRate = winRatePercent / 100.0;


            // Using the constants defined at the top for remaining hidden parameters

            const totalQuotesPerDayCurrent = numQuoters * quotesPerDayPerQuoter;

            const currentQuotesPerMonth = totalQuotesPerDayCurrent * BUSINESS_DAYS_PER_MONTH;

            const additionalQuotesPerMonth = currentQuotesPerMonth * (EFFICIENCY_INC_LEARNING_CURVE - 1.0);

            const avgProjectValue = linesPerQuote * avgProductValuePerLine; 

            const netNewSalesPerMonth = additionalQuotesPerMonth * winRate * avgProjectValue; 

            const newGrossProfitPerMonth = netNewSalesPerMonth * (approxGrossMarginPercent / 100.0);

            

            const totalMonthlyAqFees = FIXED_TOTAL_MONTHLY_AQ_FEES; // Use the constant for calculations

            const netMonthlyBenefit = newGrossProfitPerMonth - totalMonthlyAqFees;

            

            let fullYearROIPercent = 0;

            const totalAnnualNetBenefit = (newGrossProfitPerMonth * 12) - (totalMonthlyAqFees * 12);

            const totalAnnualCost = totalMonthlyAqFees * 12;


            if (totalAnnualCost > 0) {

                fullYearROIPercent = (totalAnnualNetBenefit / totalAnnualCost) * 100;

            } else if (totalAnnualNetBenefit > 0) {

                fullYearROIPercent = Infinity; 

            }


            // --- Display Monthly Results ---

            document.getElementById('net_new_sales_per_month').textContent = formatCurrency(netNewSalesPerMonth);

            document.getElementById('new_gross_profit_per_month').textContent = formatCurrency(newGrossProfitPerMonth);

            document.getElementById('total_monthly_aq_fees').textContent = formatCurrency(totalMonthlyAqFees); // Display the fixed fees

            document.getElementById('net_monthly_benefit').textContent = formatCurrency(netMonthlyBenefit);

            document.getElementById('full_year_roi_percent').textContent = fullYearROIPercent === Infinity ? "Infinite" : formatPercentage(fullYearROIPercent);


            // --- Prepare Data for Quarterly Cumulative Chart ---

            let cumulativeCostsData = [];

            let cumulativeGrossProfitData = [];

            let currentCumulativeCost = 0;

            let currentCumulativeGrossProfit = 0;

            const quarterlyAqFees = totalMonthlyAqFees * 3;

            const quarterlyNewGrossProfit = newGrossProfitPerMonth * 3;


            for (let q = 0; q < PROJECTION_QUARTERS; q++) {

                currentCumulativeCost += quarterlyAqFees;

                cumulativeCostsData.push(currentCumulativeCost);


                if (q === 0) { // First quarter, profit is delayed

                    cumulativeGrossProfitData.push(0);

                } else {

                    currentCumulativeGrossProfit += quarterlyNewGrossProfit;

                    cumulativeGrossProfitData.push(currentCumulativeGrossProfit);

                }

            }

            createOrUpdateCumulativeChart(cumulativeCostsData, cumulativeGrossProfitData);

        }


        // --- Event Listener ---

        document.getElementById('calculate_button').addEventListener('click', calculateROI);


        // --- Initial Page Load ---

        window.onload = function() {

            // Set the fixed monthly fees in the assumptions table dynamically

            document.getElementById('fixed_monthly_fees_assumption_value').textContent = formatCurrency(FIXED_TOTAL_MONTHLY_AQ_FEES);

            // Also update the display in the results section (though it's also updated in calculateROI)

            document.getElementById('total_monthly_aq_fees').textContent = formatCurrency(FIXED_TOTAL_MONTHLY_AQ_FEES);

            

            calculateROI(); // Calculate with default values when the page loads

        };
